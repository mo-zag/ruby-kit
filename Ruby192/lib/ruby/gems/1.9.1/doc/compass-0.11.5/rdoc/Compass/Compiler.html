 84:     def run
 85:       failure_count = 0
 86:       if new_config?
 87:         # Wipe out the cache and force compilation if the configuration has changed.
 88:         remove options[:cache_location] if options[:cache_location]
 89:         options[:force] = true
 90:       end
 91: 
 92:       # Make sure the target directories exist
 93:       target_directories.each {|dir| directory dir}
 94: 
 95:       # Compile each sass file.
 96:       result = timed do
 97:         sass_files.zip(css_files).each do |sass_filename, css_filename|
 98:           begin
 99:             compile_if_required sass_filename, css_filename
100:           rescue Sass::SyntaxError => e
101:             failure_count += 1
102:             handle_exception(sass_filename, css_filename, e)
103:           end
104:         end
105:       end
106:       if options[:time]
107:         puts "Compilation took #{(result.__duration * 1000).round / 1000.0}s"
108:       end
109:       return failure_count
110:     end
