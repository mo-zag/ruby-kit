394:     def normalize_stops(color_list)
395:       positions = color_list.value.map{|obj| obj.dup}
396:       # fill in the start and end positions, if unspecified
397:       positions.first.stop = Sass::Script::Number.new(0) unless positions.first.stop
398:       positions.last.stop = Sass::Script::Number.new(100, ["%"]) unless positions.last.stop
399:       # fill in empty values
400:       for i in 0...positions.size
401:         if positions[i].stop.nil?
402:           num = 2.0
403:           for j in (i+1)...positions.size
404:             if positions[j].stop
405:               positions[i].stop = positions[i-1].stop.plus((positions[j].stop.minus(positions[i-1].stop)).div(Sass::Script::Number.new(num)))
406:               break
407:             else
408:               num += 1
409:             end
410:           end
411:         end
412:       end
413:       # normalize unitless numbers
414:       positions.each do |pos|
415:         if pos.stop.unitless? && pos.stop.value <= 1
416:           pos.stop = pos.stop.times(Sass::Script::Number.new(100, ["%"]))
417:         elsif pos.stop.unitless?
418:           pos.stop = pos.stop.times(Sass::Script::Number.new(1, ["px"]))
419:         end
420:       end
421:       if (positions.last.stop.eq(Sass::Script::Number.new(0, ["px"])).to_bool ||
422:          positions.last.stop.eq(Sass::Script::Number.new(0, ["%"])).to_bool)
423:          raise Sass::SyntaxError.new("Color stops must be specified in increasing order")
424:        end
425:       if defined?(Sass::Script::List)
426:         Sass::Script::List.new(positions, color_list.separator)
427:       else
428:         color_list.class.new(*positions)
429:       end
430:     end
