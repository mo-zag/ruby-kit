246:       def process_result
247:         if @options[:interactive]
248:           require 'sass'
249:           require 'sass/repl'
250:           ::Sass::Repl.new(@options).run
251:           return
252:         end
253: 
254:         super
255: 
256:         begin
257:           input = @options[:input]
258:           output = @options[:output]
259: 
260:           tree =
261:             if input.is_a?(File) && !@options[:check_syntax]
262:               ::Sass::Files.tree_for(input.path, @options[:for_engine])
263:             else
264:               # We don't need to do any special handling of @options[:check_syntax] here,
265:               # because the Sass syntax checking happens alongside evaluation
266:               # and evaluation doesn't actually evaluate any code anyway.
267:               ::Sass::Engine.new(input.read(), @options[:for_engine]).to_tree
268:             end
269: 
270:           input.close() if input.is_a?(File)
271: 
272:           output.write(tree.render)
273:           output.close() if output.is_a? File
274:         rescue ::Sass::SyntaxError => e
275:           raise e if @options[:trace]
276:           raise "Syntax error on line #{get_line e}: #{e.message}"
277:         end
278:       end
