 82:     def compile_template(handler, template, file_name, local_assigns)
 83:       render_symbol = assign_method_name(handler, template, file_name)
 84: 
 85:       # Move begin up two lines so it captures compilation exceptions.
 86:       begin
 87:         render_source = create_template_source(handler, template, render_symbol, local_assigns.keys)
 88:         line_offset = @@template_args[render_symbol].size + handler.line_offset
 89:       
 90:         file_name = 'compiled-template' if file_name.blank?
 91:         CompiledTemplates.module_eval(render_source, file_name, -line_offset)
 92:       rescue Exception => e # errors from template code
 93:         if logger
 94:           logger.debug "ERROR: compiling #{render_symbol} RAISED #{e}"
 95:           logger.debug "Function body: #{render_source}"
 96:           logger.debug "Backtrace: #{e.backtrace.join("\n")}"
 97:         end
 98: 
 99:         # There's no way to tell Haml about the filename,
100:         # so we've got to insert it ourselves.
101:         e.backtrace[0].gsub!('(haml)', file_name) if e.is_a?(Haml::Error)
102:         
103:         raise ActionView::TemplateError.new(extract_base_path_from(file_name) || view_paths.first, file_name || template, @assigns, template, e)
104:       end
105:       
106:       @@compile_time[render_symbol] = Time.now
107:       # logger.debug "Compiled template #{file_name || template}\n ==> #{render_symbol}" if logger
108:     end
