This is a higher speed (if messier) version of wrap 

     # File lib/rdoc/markup/to_latex.rb, line 150
150:   def wrap(txt, line_len = 76)
151:     res = ""
152:     sp = 0
153:     ep = txt.length
154:     while sp < ep
155:       # scan back for a space
156:       p = sp + line_len - 1
157:       if p >= ep
158:         p = ep
159:       else
160:         while p > sp and txt[p] != \s\
161:           p -= 1
162:         end
163:         if p <= sp
164:           p = sp + line_len
165:           while p < ep and txt[p] != \s\
166:             p += 1
167:           end
168:         end
169:       end
170:       res << txt[sp...p] << "\n"
171:       sp = p
172:       sp += 1 while sp < ep and txt[sp] == \s\
173:     end
174:     res
175:   end
