RDoc::Context

523:       result = nil
524: 
525:       case symbol
526:       when /^::(.*)/ then
527:         result = toplevel.find_symbol($1)
528:       when /::/ then
529:         modules = symbol.split(/::/)
530:

533:           result = find_module_named(module_name)
